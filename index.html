// The Daily Water – Vercel Serverless Function // Guardian API (primär) + GNews (backup) + Claude Zusammenfassung export const config = { maxDuration: 60 }; const GUARDIAN = "https://content.guardianapis.com/search"; const GNEWS = "https://gnews.io/api/v4/search"; const CATS = { aalen: { guardian: { q:"Aalen Germany" }, gnews: { q:"Aalen", lang:"de" }, ctx:"Lokalnachrichten Aalen, Ostalbkreis, Baden-Württemberg" }, deutschland: { guardian: { q:"Germany", section:"world" }, gnews: { q:"Germany", lang:"en" }, ctx:"Wichtigste Nachrichten aus Deutschland" }, welt: { guardian: { q:"world news", section:"world" }, gnews: { q:"world", lang:"en" }, ctx:"Wichtigste weltweite Nachrichten" }, water4you: { guardian: { q:"water supply drinking water environment" }, gnews: { q:"water", lang:"en" }, ctx:"Wasserversorgung, Trinkwasser – Water 4 You GmbH Aalen" }, zielmaerkte: { guardian: { q:"water infrastructure Middle East Africa" }, gnews: { q:"infrastructure", lang:"en" }, ctx:"Wasser-Infrastrukturprojekte Naher Osten, Afrika, Asien, Lateinamerika" }, philippinen: { guardian: { q:"Philippines", section:"world" }, gnews: { q:"Philippines", lang:"en" }, ctx:"Aktuelle Nachrichten Philippinen" }, business: { guardian: { q:"business economy", section:"business" }, gnews: { q:"business", lang:"en" }, ctx:"Wirtschaft, Business, Märkte" }, ki_tech: { guardian: { q:"artificial intelligence technology", section:"technology" }, gnews: { q:"technology", lang:"en" }, ctx:"KI und Technologie" }, aktien: { guardian: { q:"stock market finance", section:"business" }, gnews: { q:"stocks", lang:"en" }, ctx:"Aktienmarkt, Börse, DAX, Finanzen" }, }; async function fromGuardian(key, p) { try { const params = new URLSearchParams({ "api-key":key, "show-fields":"thumbnail,trailText", "page-size":"8", "order-by":"newest", ...p }); const r = await fetch(GUARDIAN + "?" + params); const d = await r.json(); if (!r.ok || d.response?.status !== "ok") return []; return (d.response?.results || []).filter(a => a.webTitle && a.fields?.trailText).slice(0,6).map(a => ({ title: a.webTitle, description: a.fields.trailText, image: a.fields?.thumbnail || "", url: a.webUrl, source: "The Guardian", date: (a.webPublicationDate||"").slice(0,10) })); } catch { return []; } } async function fromGNews(key, p) { try { const params = new URLSearchParams({ apikey:key, max:"8", sortby:"publishedAt", ...p }); const r = await fetch(GNEWS + "?" + params); const d = await r.json(); if (!r.ok || d.errors) return []; return (d.articles||[]).filter(a => a.title && a.description && !a.title.includes("[Removed]")).slice(0,6).map(a => ({ title: a.title, description: a.description, image: a.image||"", url: a.url, source: a.source?.name||"GNews", date: (a.publishedAt||"").slice(0,10) })); } catch { return []; } } async function summarise(apiKey, articles, ctx, attempt=0) { if (attempt > 2) throw new Error("Claude Rate-Limit. Bitte 1 Minute warten."); if (attempt > 0) await new Promise(r => setTimeout(r, 5000 * attempt)); const input = articles.map((a,i) => `[${i+1}] ${a.title}\nQuelle: ${a.source} | ${a.date}\n${a.description}\nBild: ${a.image}\nURL: ${a.url}` ).join("\n\n"); const r = await fetch("https://api.anthropic.com/v1/messages", { method:"POST", headers:{ "Content-Type":"application/json", "x-api-key":apiKey, "anthropic-version":"2023-06-01" }, body: JSON.stringify({ model:"claude-haiku-4-5-20251001", max_tokens:3000, system:"Antworte NUR mit einem JSON-Array. Kein Text davor/danach. Keine Codeblöcke. Nur [ ... ].", messages:[{ role:"user", content: `Kontext: ${ctx}\n\n${articles.length} echte Artikel:\n\n${input}\n\n` + `JSON-Array auf Deutsch:\n[{"title":"Titel max 12 Wörter","category":"Thema","source":"Quelle","date":"TT.MM.JJJJ","summary":"2-3 Sätze.","fullText":"6-8 Sätze.","relevance":"hoch","imageUrl":"Bild-URL","articleUrl":"Artikel-URL"}]` }] }) }); const d = await r.json(); if (r.status === 429) return summarise(apiKey, articles, ctx, attempt+1); if (!r.ok) throw new Error("Claude " + r.status + ": " + (d?.error?.message||"Fehler")); const raw = (d.content||[]).filter(b=>b.type==="text").map(b=>b.text).join("").trim(); const s = raw.indexOf("["), e = raw.lastIndexOf("]"); if (s<0||e<=s) throw new Error("Kein JSON. Claude: \""+raw.slice(0,150)+"\""); const result = JSON.parse(raw.slice(s,e+1)); if (!Array.isArray(result)||!result.length) throw new Error("Leeres Array."); return result; } export default async function handler(req, res) { res.setHeader("Access-Control-Allow-Origin","*"); res.setHeader("Access-Control-Allow-Methods","POST, OPTIONS"); res.setHeader("Access-Control-Allow-Headers","Content-Type"); if (req.method==="OPTIONS") { res.status(200).end(); return; } if (req.method!=="POST") { res.status(405).end(); return; } const AKEY = process.env.ANTHROPIC_API_KEY; const GKEY = process.env.GUARDIAN_API_KEY; const NKEY = process.env.GNEWS_API_KEY; if (!AKEY) { res.status(500).json({error:{message:"ANTHROPIC_API_KEY fehlt in Vercel → Settings → Environment Variables"}}); return; } const { category } = req.body||{}; const cat = CATS[category]; if (!cat) { res.status(400).json({error:{message:"Unbekannte Kategorie: "+category}}); return; } try { let articles = [], source = ""; if (GKEY) { articles = await fromGuardian(GKEY, cat.guardian); if (articles.length) source = "guardian"; } if (!articles.length && NKEY) { articles = await fromGNews(NKEY, cat.gnews); if (articles.length) source = "gnews"; } if (!articles.length) { const missing = []; if (!GKEY) missing.push("GUARDIAN_API_KEY"); if (!NKEY) missing.push("GNEWS_API_KEY"); throw new Error(missing.length ? "Fehlende Keys in Vercel: " + missing.join(" und/oder ") + " → Settings → Environment Variables" : "Keine Artikel von News-APIs erhalten." ); } const result = await summarise(AKEY, articles, cat.ctx); res.status(200).json({ articles: result, source }); } catch(err) { console.error("[daily-water]", category, err.message); res.status(500).json({error:{message:err.message}}); } }
